2.ReadAndFindMinSalaryOfNDoctors() without array

PSEUDO ROUTINE:
ReadAndFindMinSalaryOfNDoctors(INT n)
DECLARE INT min = -1
DECLARE INT salary = 0
FOR INT i FROM 0 TO n
	PRINT "Enter 'i' th doctor's salary"
	INPUT INT salary
	IF (salary < min)
		min = salary
	END IF
	salary = 0 
END FOR
RETURN min

PSEUDO ROUTINE:
ReadAndFindSumOfDoctorsSalariesTillZero()
DECLARE INT sum = 0
PRINT "Enter 1st salary"
INPUT INT salary
DO
	IF (salary < 0)
		PRINT "Enter valid salary"
		INPUT salary
		CONTINUE
	ELSE IF (salary = 0)
		BREAK
	ELSE
		sum += salary
	END IF
	PRINT "Enter next salary"
	INPUT salary
WHILE (salary != 0)
END WHILE
RETURN sum
